// 1) Raise the canvas above any overlays
(() => { const c = document.getElementById('game'); if (c) { c.style.position='relative'; c.style.zIndex='10'; c.style.display='block'; }} )();

// 2) Demote overlays so they cannot occlude the canvas
(() => { document.querySelectorAll('.overlay,.vignette,.frame,[data-role="overlay"]').forEach(el => { el.style.pointerEvents='none'; el.style.zIndex='1'; }); })();

// 3) Ensure HUD won’t block input (still visible)
(() => { const h = document.getElementById('hudTL'); if (h) { h.style.position='absolute'; h.style.pointerEvents='none'; h.style.zIndex='20'; h.style.background='transparent'; }} )();
